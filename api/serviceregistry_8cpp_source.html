42 void libcurl_perform_checked_throw(HTTP::CURLContext& ctx)
43 {
44  /* Perform the request */
45  CURLcode curl_code = curl_easy_perform(ctx.curl);
46 
47  /* Check for errors */
48  if (curl_code != CURLE_OK) {
49  std::ostringstream ss;
50  ss << "CURLError " << curl_code <<
51  ": " << curl_easy_strerror(curl_code) <<
52  ", " << ctx.errbuf;
53  throw TransportError(ss.str());
54  }
55  long http_code = 0;
56  curl_easy_getinfo (ctx.curl, CURLINFO_RESPONSE_CODE, &http_code);
57  if (http_code != 200)
58  {
59  std::ostringstream ss;
60  ss << "HTTPError " << http_code;
61  /* Fail */
62  throw TransportError(ss.str());
63  }
64 }
